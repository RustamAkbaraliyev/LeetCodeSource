 public  int[] twoSum(int[] nums, int target) {
        int[] array = new int[2];
        label:
        for (int i = 0; i < nums.length; i++) {
            for (int j = i + 1; j < nums.length; j++) {

                if (nums[i] + nums[j] == target) {
                    array[0] = i;
                    array[1] = j;
                    break label;
                }
            }
        }

        return array;
    }
